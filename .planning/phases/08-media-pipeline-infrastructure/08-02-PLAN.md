```typescript
/**
 * Validates HMAC-SHA256 signature on incoming B2 Event Notification webhooks.
 * B2 sends the signature in the X-Bz-Event-Notification-Signature header
 * as "v1={hex-encoded-hmac-sha256}".
 */
export async function validateHmac(
  body: string,
  signatureHeader: string | null,
  secret: string
): Promise<boolean> {
  if (!signatureHeader) return false

  // B2 signature format: "v1={hex}"
  const match = signatureHeader.match(/^v1=([a-f0-9]+)$/i)
  if (!match) return false
  const receivedHex = match[1]

  const encoder = new TextEncoder()
  const key = await crypto.subtle.importKey(
    'raw',
    encoder.encode(secret),
    { name: 'HMAC', hash: 'SHA-256' },
    false,
    ['sign']
  )

  const signature = await crypto.subtle.sign('HMAC', key, encoder.encode(body))
  const expectedHex = Array.from(new Uint8Array(signature))
    .map((b) => b.toString(16).padStart(2, '0'))
    .join('')

  // Constant-time comparison to prevent timing attacks
  if (receivedHex.length !== expectedHex.length) return false
  let mismatch = 0
  for (let i = 0; i < receivedHex.length; i++) {
    mismatch |= receivedHex.charCodeAt(i) ^ expectedHex.charCodeAt(i)
  }
  return mismatch === 0
}
```

**src/sanity.ts:**
```typescript
/**
 * Creates a draft video document in Sanity via the Mutations API.
 * Uses raw fetch (no @sanity/client) to minimize Worker bundle size.
 */

const SANITY_PROJECT_ID = 'fo6n8ceo'
const SANITY_DATASET = 'production'
const SANITY_API_VERSION = '2024-01-01'

export interface VideoFields {
  title: string
  b2Key: string
  cdnUrl: string
  videoSource: 'b2'
  bunnyStatus: 'processing' | 'ready' | 'error'
}

export async function createVideoDraft(
  fields: VideoFields,
  sanityToken: string
): Promise<{ documentId: string; success: boolean; error?: string }> {
  const documentId = `drafts.${crypto.randomUUID()}`

  const body = {
    mutations: [
      {
        create: {
          _id: documentId,
          _type: 'video',
          title: fields.title,
          videoSource: fields.b2Key ? 'b2' : 'wistia',
          b2Key: fields.b2Key,
          cdnUrl: fields.cdnUrl,
          bunnyStatus: fields.bunnyStatus,
          // Set required defaults so editors only need to fill in editorial fields
          narrativeOwner: 'hector',
          platformTier: 'canonical',
          archivalStatus: 'archival',
          videoFormat: 'longform',
          language: ['en'],
        },
      },
    ],
  }

  const url = `https://${SANITY_PROJECT_ID}.api.sanity.io/v${SANITY_API_VERSION}/data/mutate/${SANITY_DATASET}`

  try {
    const response = await fetch(url, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        Authorization: `Bearer ${sanityToken}`,
      },
      body: JSON.stringify(body),
    })

    if (!response.ok) {
      const errorText = await response.text()
      return { documentId, success: false, error: `Sanity API ${response.status}: ${errorText}` }
    }

    return { documentId, success: true }
  } catch (err) {
    return { documentId, success: false, error: `Fetch failed: ${(err as Error).message}` }
  }
}
```

**src/index.ts:**
```typescript
import { validateHmac } from './hmac'
import { createVideoDraft } from './sanity'
import type { VideoFields } from './sanity'

interface Env {
  SANITY_TOKEN: string
  B2_HMAC_SECRET: string
  BUNNY_CDN_HOSTNAME: string
}

interface B2EventRecord {
  s3: {
    bucket: { name: string }
    object: { key: string; size: number }
  }
  eventName: string
}

interface B2EventPayload {
  Records: B2EventRecord[]
}

/**
 * Extracts a human-readable title from a B2 object key.
 * "videos/2026/my-keynote-speech.mp4" -> "my-keynote-speech"
 */
function titleFromKey(key: string): string {
  const filename = key.split('/').pop() || key
  // Remove file extension
  return filename.replace(/\.[^/.]+$/, '').replace(/[-_]/g, ' ')
}

export default {
  async fetch(request: Request, env: Env): Promise<Response> {
    // Only accept POST requests
    if (request.method !== 'POST') {
      return new Response('Method not allowed', { status: 405 })
    }

    // Read body for HMAC validation
    const body = await request.text()

    // Validate HMAC signature
    const signature = request.headers.get('X-Bz-Event-Notification-Signature')
    const isValid = await validateHmac(body, signature, env.B2_HMAC_SECRET)
    if (!isValid) {
      console.error('HMAC validation failed')
      return new Response('Unauthorized', { status: 401 })
    }

    // Parse B2 event payload
    let payload: B2EventPayload
    try {
      payload = JSON.parse(body)
    } catch {
      return new Response('Invalid JSON', { status: 400 })
    }

    if (!payload.Records || payload.Records.length === 0) {
      return new Response('No records in event', { status: 400 })
    }

    const results: Array<{ key: string; documentId: string; success: boolean; error?: string }> = []

    for (const record of payload.Records) {
      const objectKey = record.s3.object.key

      // Only process video files in the videos/ prefix
      if (!objectKey.startsWith('videos/')) {
        console.log(`Skipping non-video object: ${objectKey}`)
        continue
      }

      // Only process common video file extensions
      const ext = objectKey.split('.').pop()?.toLowerCase()
      if (!ext || !['mp4', 'mov', 'webm', 'mkv', 'avi'].includes(ext)) {
        console.log(`Skipping non-video file extension: ${objectKey}`)
        continue
      }

      const cdnUrl = `https://${env.BUNNY_CDN_HOSTNAME}/${objectKey}`
      const title = titleFromKey(objectKey)

      const fields: VideoFields = {
        title,
        b2Key: objectKey,
        cdnUrl,
        videoSource: 'b2',
        bunnyStatus: 'processing',
      }

      const result = await createVideoDraft(fields, env.SANITY_TOKEN)
      results.push({ key: objectKey, ...result })

      if (!result.success) {
        console.error(`Failed to create draft for ${objectKey}: ${result.error}`)
      } else {
        console.log(`Created draft ${result.documentId} for ${objectKey}`)
      }
    }

    return new Response(JSON.stringify({ processed: results }), {
      status: 200,
      headers: { 'Content-Type': 'application/json' },
    })
  },
}
```

**scripts/validate-cors.sh:**
```bash
#!/usr/bin/env bash
# Validates B2 CORS rules allow Bunny CDN origin
# Usage: ./scripts/validate-cors.sh <b2-bucket-url> <bunny-cdn-hostname>

set -euo pipefail

B2_URL="${1:?Usage: validate-cors.sh <b2-bucket-url> <bunny-cdn-hostname>}"
CDN_HOST="${2:?Usage: validate-cors.sh <b2-bucket-url> <bunny-cdn-hostname>}"

echo "Testing CORS from origin https://${CDN_HOST}..."

RESPONSE=$(curl -sI \
  -H "Origin: https://${CDN_HOST}" \
  -H "Access-Control-Request-Method: GET" \
  -X OPTIONS \
  "${B2_URL}" 2>&1)

if echo "$RESPONSE" | grep -qi "access-control-allow-origin"; then
  echo "PASS: CORS headers present"
  echo "$RESPONSE" | grep -i "access-control"
else
  echo "FAIL: No CORS headers returned"
  echo "$RESPONSE"
  exit 1
fi
```

```bash
#!/usr/bin/env bash
# Validates Bunny CDN pull zone is correctly configured with B2 origin
# Usage: ./scripts/validate-cdn.sh <bunny-cdn-hostname>

set -euo pipefail

CDN_HOST="${1:?Usage: validate-cdn.sh <bunny-cdn-hostname>}"

echo "Testing Bunny CDN pull zone at ${CDN_HOST}..."

# Test 1: Root path should be blocked (Block Root Path Access enabled)
ROOT_STATUS=$(curl -sI -o /dev/null -w "%{http_code}" "https://${CDN_HOST}/")
if [ "$ROOT_STATUS" = "403" ] || [ "$ROOT_STATUS" = "404" ]; then
  echo "PASS: Root path blocked (HTTP ${ROOT_STATUS})"
else
  echo "WARN: Root path returned HTTP ${ROOT_STATUS} (expected 403 or 404)"
fi

# Test 2: Non-existent video path should return 404 (not 403)
VIDEO_STATUS=$(curl -sI -o /dev/null -w "%{http_code}" "https://${CDN_HOST}/videos/nonexistent-test-file.mp4")
if [ "$VIDEO_STATUS" = "404" ]; then
  echo "PASS: Non-existent video returns 404 (CDN is reaching B2 origin)"
else
  echo "FAIL: Non-existent video returned HTTP ${VIDEO_STATUS} (expected 404, got ${VIDEO_STATUS} — check B2 origin config)"
  exit 1
fi

echo ""
echo "CDN validation complete."
```
